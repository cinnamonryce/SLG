on_conquest = {
    if = {
        limit = {
			has_province_modifier = emerald_buddha
            owner = {
                NOT = {
                    religion = buddhism
                    secondary_religion = buddhism
                    religion = vajrayana
                    secondary_religion = vajrayana
                    religion = mahayana
                    secondary_religion = mahayana
                }
            }
        }
		remove_province_modifier = emerald_buddha
		add_province_modifier = {
			name = "emerald_buddha_heathen"
			duration = -1
		}
	}
}

		is_year = 1444
		is_subject  = no

			any_owned_province = {
				any_neighbor_province = {
					owned_by = MMA
				}
			}
			any_subject_country = {
				any_owned_province = {
					any_neighbor_province = {
						owned_by = MMA
					}
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 60
	}
	
	option = {
		name = "mma.1.a" #Their banditry ends now!
		ai_chance = {
			factor = 100
			modifier = {
				factor = 0
				OR = {
					is_bankrupt = yes
					war_exhaustion = 10
				}
			}
		}
		declare_war_with_cb = {
			who = MMA
			casus_belli = cb_chinese_unification
		}
	}
	option = {
		name = mma.1.b #We cannot afford any more losses
		add_prestige = -25
	}
}

#The Fall of Mong Mao
country_event = {
	id = mma.2
	title = "mma.2.t"
	desc = "mma.2.d"
	picture = BATTLE_eventPicture

	fire_only_once = yes	
	
	trigger = {
		MMA = {
			ai = yes
		}
		OR = {
			AND = {
				tag = MNG
				NOT = { exists = QNG }
			}
			tag = QNG
		}
		2166 = {
			owned_by = MMA
		}
		2166 = {
			controlled_by = ROOT
		}
	}

	immediate = {
		hidden_effect = {
			random_subject_country = {
				limit = {
					any_owned_province = {
						any_neighbor_province = {
							province_id = 2166
						}
					}
					NOT = {
						is_subject_of_type = tributary_state
					}
				}
				save_event_target_as = yunnan_subject
			}
		}
	}
	
	option = {
		name = "mma.2.a" #We stand triumphant!
		if = {
			limit = {
				any_subject_country = {
					any_owned_province = {
						any_neighbor_province = {
							province_id = 2166
						}
					}
				}
			}
			2166 = { cede_province = event_target:yunnan_subject }	
		}
		else = {
			2166 = {
				cede_province = ROOT
			}
		}
		MMA = { 
			change_tag = MYA
		}
		if = {
			limit = {
				MKA = {
					OR = {
						exists = no
						is_subject_of = MYA
					}
				}
			}
			MYA = {
				release = MKA
			}
			create_subject = {
				subject_type = tributary_state
				subject = MKA
			}
		}
		white_peace = MYA
		add_truce_with = MYA
		add_truce_with = MKA
		hidden_effect = {
			MYA = { 
				set_capital = 582
			}	
		}
		create_subject = {
			subject_type = tributary_state
			subject = MYA
		}
	}

	option = {
		name = mma.2.b #We are not finished with this war
		ai_chance = {
			factor = 0
		}
		add_mil_power = -50
		add_war_exhaustion = 2
	}
}
